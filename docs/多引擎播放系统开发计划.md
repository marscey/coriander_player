# 多引擎播放系统开发计划

## 一、项目概述

基于 BASS 库的播放器引擎在播放在线音频（流式播放）方面存在不足，本开发计划旨在引入 media_kit 库构建新的播放引擎，同时保留 BASS 库的实现方案，支持通过配置设置切换不同的播放引擎。默认情况下，桌面端（Windows、macOS）采用 BASS 库实现，移动端（Android、iOS）采用 media_kit 库实现。

## 二、开发阶段与任务分解

### 第一阶段：架构设计与基础组件实现（2周）

#### 1.1 定义播放器引擎接口 (PlayerEngine)
- **任务描述**：创建统一的播放器引擎接口，规定所有播放器引擎必须实现的方法
- **优先级**：高
- **预计工时**：4小时
- **交付物**：`lib/play_service/engine/player_engine.dart`

```dart
// 定义 PlayerEngine 接口
export 'package:coriander_player/play_service/bass_player.dart';

abstract class PlayerEngine {
  // 初始化播放器
  Future<void> initialize();
  
  // 设置音频源
  Future<void> setSource(String path, {bool isAsset = false, bool isNetwork = false});
  
  // 播放控制
  Future<void> play();
  Future<void> pause();
  Future<void> stop();
  Future<void> seek(Duration position);
  
  // 音频属性控制
  void setVolume(double volume);
  void setSpeed(double speed);
  
  // 状态获取
  PlayerState get state;
  Duration get position;
  Duration get duration;
  
  // 状态流
  Stream<PlayerState> get playerStateStream;
  Stream<Duration> get positionStream;
  
  // 释放资源
  Future<void> dispose();
}
```

```dart
import 'dart:io';

import 'package:flutter/foundation.dart';
import 'package:flutter/material.dart';

import 'package:coriander_player/app_settings.dart';
import 'package:coriander_player/play_service/engine/bass_player_engine.dart';
import 'package:coriander_player/play_service/engine/media_kit_player_engine.dart';
import 'package:coriander_player/play_service/engine/player_engine.dart';
import 'package:coriander_player/play_service/engine/player_engine_type.dart';

class PlayerEngineFactory {
  static PlayerEngine createEngine(PlayerEngineType type) {
    switch (type) {
      case PlayerEngineType.bass:
        return BassPlayerEngine();
      case PlayerEngineType.mediaKit:
          return MediaKitPlayerEngine();
      default:
        throw UnimplementedError('Unsupported player engine type');
    }
  }
  
  // 根据平台和配置获取默认引擎
  static PlayerEngine getDefaultEngine() {
    final config = AppSettings.instance;
    final engineType = config.playerEngineType;
    
    if (engineType != null) {
      return createEngine(engineType);
    }
    
    // 根据平台选择默认引擎
    switch (defaultTargetPlatform) {
      case TargetPlatform.android:
      case TargetPlatform.iOS:
        return createEngine(PlayerEngineType.mediaKit);
      case TargetPlatform.windows:
      case TargetPlatform.macOS:
      default:
        return createEngine(PlayerEngineType.bass);
    }
  }
}
```

#### 2.2 实现 media_kit 引擎适配器
- **任务描述**：创建新的 MediaKitPlayerEngine 类，基于 media_kit 库实现 PlayerEngine 接口
- **优先级**：高
- **预计工时**：12小时
- **交付物**：`lib/play_service/engine/media_kit_player_engine.dart`

```dart
import 'package:media_kit/media_kit.dart';
import 'package:coriander_player/play_service/bass_player.dart';
import 'package:coriander_player/play_service/engine/player_engine.dart';

class MediaKitPlayerEngine implements PlayerEngine {
  late final AudioPlayer _player;
  late final StreamController<PlayerState> _playerStateStreamController;
  late final StreamController<Duration> _positionStreamController;
  late final Duration _duration;
  Timer? _positionTimer;
  
  @override
  Future<void> initialize() async {
    _player = AudioPlayer();
    _playerStateStreamController = StreamController<PlayerState>.broadcast();
    _positionStreamController = StreamController<Duration>.broadcast();
    _duration = Duration.zero;
    
    // 监听播放状态变化
    _player.onPlayerStateChanged.listen((state) {
      PlayerState playerState;
      switch (state) {
        case PlayerState.playing:
          playerState = PlayerState.playing;
          break;
        case PlayerState.paused:
          playerState = PlayerState.paused;
          break;
        case PlayerState.stopped:
          playerState = PlayerState.stopped;
          break;
        default:
          playerState = PlayerState.stopped;
      }
      _playerStateStreamController.add(playerState);
    });
    
    // 监听播放位置变化
    _player.onPositionChanged.listen((position) {
      _positionStreamController.add(position);
    });
    
    // 监听播放完成
    _player.onPlayerCompletion.listen((event) {
      _playerStateStreamController.add(PlayerState.stopped);
    });
    
    // 设置循环播放模式
    await _player.setLoopMode(LoopMode.off);
  }
  
  @override
  Future<void> setSource(String path, {bool isAsset = false, bool isNetwork = false}) async {
    try {
      if (isNetwork) {
        await _player.setUrl(path);
      } else if (isAsset) {
        await _player.setAsset(path);
      } else {
        await _player.setFilePath(path);
      }
      
      // 获取音频时长
      final duration = await _player.getDuration();
      if (duration != null) {
        _duration = Duration(milliseconds: duration);
      }
    } catch (e) {
      print('Error setting source: $e');
      throw e;
    }
  }
  
  @override
  Future<void> play() {
    return _player.play();
  }
  
  @override
  Future<void> pause() {
    return _player.pause();
  }
  
  @override
  Future<void> stop() {
    return _player.stop();
  }
  
  @override
  Future<void> seek(Duration position) {
    return _player.seek(position);
  }
  
  @override
  void setVolume(double volume) {
    _player.setVolume(volume);
  }
  
  @override
  void setSpeed(double speed) {
    _player.setPlaybackRate(speed);
  }
  
  @override
  PlayerState get state {
    switch (_player.playerState) {
      case PlayerState.playing:
        return PlayerState.playing;
      case PlayerState.paused:
        return PlayerState.paused;
      case PlayerState.stopped:
        return PlayerState.stopped;
      default:
        return PlayerState.stopped;
    }
  }
  
  @override
  Duration get position {
    return _player.currentPosition ?? Duration.zero;
  }
  
  @override
  Duration get duration {
    return _duration;
  }
  
  @override
  Stream<PlayerState> get playerStateStream {
    return _playerStateStreamController.stream;
  }
  
  @override
  Stream<Duration> get positionStream {
    return _positionStreamController.stream;
  }
  
  @override
  Future<void> dispose() async {
    _positionTimer?.cancel();
    await _playerStateStreamController.close();
    await _positionStreamController.close();
    await _player.dispose();
  }
}
```
